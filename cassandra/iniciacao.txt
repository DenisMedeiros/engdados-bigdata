## Cria o keyspace (banco de dados)

CREATE KEYSPACE eleicoes WITH replication = {'class': 'SimpleStrategy', 'replication_factor' : 1};

## Cria as tabelas

# Tabela resultados2014

CREATE TABLE eleicoes.resultados2014(id UUID PRIMARY KEY, DATA_GERACAO date, HORA_GERACAO time, ANO_ELEICAO int, NUM_TURNO int, DESCRICAO_ELEICAO text, SIGLA_UF text, SIGLA_UE text, CODIGO_MUNICIPIO int, NOME_MUNICIPIO text, NUMERO_ZONA int, CODIGO_CARGO int, NUMERO_CAND int, SQ_CANDIDATO varint, NOME_CANDIDATO text, NOME_URNA_CANDIDATO text, DESCRICAO_CARGO text, COD_SIT_CAND_SUPERIOR int, DESC_SIT_CAND_SUPERIOR text, CODIGO_SIT_CANDIDATO int, DESC_SIT_CANDIDATO text, NUMERO_PARTIDO int, SIGLA_PARTIDO text, NOME_PARTIDO text, SEQUENCIAL_LEGENDA varint, NOME_COLIGACAO text, COMPOSICAO_LEGENDA text, TOTAL_VOTOS int);i

# Tabela candidatos2014

CREATE TABLE eleicoes.candidatos2014(id UUID PRIMARY KEY, DATA_GERACAO date, HORA_GERACAO time, ANO_ELEICAO int, NUM_TURNO int, DESCRICAO_ELEICAO text, SIGLA_UF text, SIGLA_UE text, DESCRICAO_UE text, CODIGO_CARGO int, DESCRICAO_CARGO text, NOME_CANDIDATO text, SEQUENCIAL_CANDIDATO varint, NUMERO_CANDIDATO int, CPF_CANDIDATO text, NOME_URNA_CANDIDATO text, COD_SITUACAO_CANDIDATURA int, DES_SITUACAO_CANDIDATURA text, NUMERO_PARTIDO int, SIGLA_PARTIDO text, NOME_PARTIDO text, CODIGO_LEGENDA varint, SIGLA_LEGENDA text, COMPOSICAO_LEGENDA text, NOME_LEGENDA text,  CODIGO_OCUPACAO int, DESCRICAO_OCUPACAO text, DATA_NASCIMENTO date, NUM_TITULO_ELEITORAL_CANDIDATO text, IDADE_DATA_ELEICAO int, CODIGO_SEXO int, DESCRICAO_SEXO text, COD_GRAU_INSTRUCAO int, DESCRICAO_GRAU_INSTRUCAO text, CODIGO_ESTADO_CIVIL int, DESCRICAO_ESTADO_CIVIL text, CODIGO_COR_RACA int, DESCRICAO_COR_RACA text,  CODIGO_NACIONALIDADE int, DESCRICAO_NACIONALIDADE text, SIGLA_UF_NASCIMENTO text, CODIGO_MUNICIPIO_NASCIMENTO int, NOME_MUNICIPIO_NASCIMENTO text, DESPESA_MAX_CAMPANHA int, COD_SIT_TOT_TURNO int, DESC_SIT_TOT_TURNO text, NM_EMAIL text);

# Tabela eleitores2014

CREATE TABLE eleicoes.eleitores2014(id UUID PRIMARY KEY, PERIODO int, UF text, MUNICIPIO text, COD_MUNICIPIO_TSE int, NR_ZONA int, SEXO text, FAIXA_ETARIA text, GRAU_DE_ESCOLARIDADE text, QTD_ELEITORES_NO_PERFIL int);

# Tabela pibs2014

CREATE TABLE eleicoes.pibs20142014(id UUID PRIMARY KEY, VALOR varint, MUNICIPIO int, ANO int);
